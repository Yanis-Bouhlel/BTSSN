Configuration sous Linux

L'objectif n'est pas de vous apprendre à installer un serveur DHCP (voir le document ressource correspondant) mais de vous donner les lignes importantes du fichier de configuration du serveur DHCP au travers d'un exemple (les lignes commençant pas # sont des commentaires).

Exemple : Exemple de fichier "/etc/dhcp/dhcpd.conf"

1
# Le nom de votre serveur dhcp
2
server-name "dhcp.monreseau.fr";
3
# Le domaine attribué à vos clients
4
option domain-name "monreseau.fr";
5
# Les serveurs DNS attribués à vos clients
6
option domain-name-servers 192.168.0.1, 192.168.0.2, 8.8.8.8;
7
# La durée des baux, en secondes (soit ici 2 heures !)
8
default-lease-time 7200;
9
max-lease-time 7200;
10
# Le masque de sous-réseau
11
option subnet-mask 255.255.255.0;
12
# L'adresse de broadcast, qui est optionnelle, mais ça vous fera un bon exercice de la calculer ;-)
13
option broadcast-address 192.168.0.255;
14
# Le routeur par défaut (passerelle)
15
option routers 192.168.0.254;
16
# Enfin, nous précisons le réseau utilisé et les particularités associées (le sous-réseau 192.168.0.0 dont le masque est 255.255.255.0)
17
subnet 192.168.0.0 netmask 255.255.255.0 {
18
# Ici, nous ne voulons attribuer dynamiquement que les adresses de 10 à 100
19
 range 192.168.0.10 192.168.0.100;
20
# Enfin, nous pouvons spécifier une adresse précise en fonction d'une machine grâce à son adresse MAC
21
# l'ordinateur qui a pour nom barbara-pc et qui a pour adresse MAC 00:23:8B:4B:D1:BD aura l'adresse IP 192.168.0.10
22
 host barbara-pc {
23
 hardware ethernet 00:23:8B:4B:D1:BD;
24
 fixed-address 192.168.0.10;
25
 }
26
}
27
# Le nom de votre serveur dhcp
server-name "dhcp.monreseau.fr";
# Le domaine attribué à vos clients
option domain-name "monreseau.fr";
# Les serveurs DNS attribués à vos clients
option domain-name-servers 192.168.0.1, 192.168.0.2, 8.8.8.8;
# La durée des baux, en secondes (soit ici 2 heures !)
default-lease-time 7200;
max-lease-time 7200;
# Le masque de sous-réseau
option subnet-mask 255.255.255.0;
# L'adresse de broadcast, qui est optionnelle, mais ça vous fera un bon exercice de la calculer ;-)
option broadcast-address 192.168.0.255;
# Le routeur par défaut (passerelle)
option routers 192.168.0.254;
# Enfin, nous précisons le réseau utilisé et les particularités associées (le sous-réseau 192.168.0.0 dont le masque est 255.255.255.0)
subnet 192.168.0.0 netmask 255.255.255.0 {
# Ici, nous ne voulons attribuer dynamiquement que les adresses de 10 à 100
 range 192.168.0.10 192.168.0.100;
# Enfin, nous pouvons spécifier une adresse précise en fonction d'une machine grâce à son adresse MAC
# l'ordinateur qui a pour nom barbara-pc et qui a pour adresse MAC 00:23:8B:4B:D1:BD aura l'adresse IP 192.168.0.10
 host barbara-pc {
 hardware ethernet 00:23:8B:4B:D1:BD;
 fixed-address 192.168.0.10;
 }
}

Remarque : la partie host est optionnelle et ne sert que quand on veut attribuer à un ordinateur une adresse IP prédéfinie (l'ordinateur est connu grâce à son adresse MAC).
